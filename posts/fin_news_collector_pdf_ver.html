def fetch_and_parse_rss(url):
    """抓取并解析指定的 RSS 提要 URL。"""
    print(f"--- 正在尝试解析 RSS 提要：{url} ---")
    feed = feedparser.parse(url, request_headers=CUSTOM_HEADERS)
    status = getattr(feed, 'status', 'N/A')
    if not feed.entries and status not in [200, 301, 302]:
        print(f"⚠️ 警告：无法从 {url} 获取文章列表。原始状态码: {status}")
        return None
    if not feed.entries:
        print(f"❌ 错误：提要对象为空，但状态码为 {status}。可能连接被拒绝或链接失效。")
        return None
    return feed

def format_published_time(published_parsed, target_timezone_name=None):
    """根据目标时区格式化发布时间为 'YYYY-MM-DD HH:MM:SS (Timezone)' 格式。"""
    if not published_parsed:
        return '无时间信息', None

    # 转换为 UTC datetime 对象
    dt_utc = datetime(*published_parsed[:6], tzinfo=pytz.utc)

    if target_timezone_name:
        try:
            target_tz = pytz.timezone(target_timezone_name)
            dt_target = dt_utc.astimezone(target_tz)
            return dt_target.strftime('%Y-%m-%d %H:%M:%S %Z'), dt_utc
        except pytz.exceptions.UnknownTimeZoneError:
            pass # 回退到 UTC

    return dt_utc.strftime('%Y-%m-%d %H:%M:%S UTC'), dt_utc

def get_ft_cutoff():
    """计算 Financial Times 的截止时间：72小时前 (UTC)。"""
    return datetime.now(pytz.utc) - timedelta(hours=72)

def get_economist_cutoff():
    """计算 Economist 的截止时间：本月和上个月的月初 (UTC)。"""
    now = datetime.now(pytz.utc)
    # 本月第一天
    first_day_current_month = now.replace(day=1, hour=0, minute=0, second=0, microsecond=0)
    # 上个月第一天：从本月第一天减去一天，再将日期设为 1
    cutoff_date = (first_day_current_month - timedelta(days=1)).replace(day=1)
    return cutoff_date

def should_keep_article(source_name, pub_dt_utc):
    """根据来源和 UTC 时间判断文章是否应保留。"""
    if pub_dt_utc is None:
        return False

    source = source_name.split(' (')[0].lower()

    # Financial Times: 72小时内
    if 'financial times' in source:
        ft_cutoff = get_ft_cutoff()
        return pub_dt_utc > ft_cutoff

    # The Economist: 本月和上个月
    elif 'economist' in source:
        eco_cutoff = get_economist_cutoff()
        return pub_dt_utc >= eco_cutoff

    # 默认保留
    return True


def process_feed_entries(feed, source_name, target_timezone_name=None, limit=None):
    """
    遍历提要中的文章，提取所需信息，应用时间筛选，并添加到 articles_data 列表中。
    """
    feed_title = feed.feed.get('title', 'N/A')
    print(f"\n## 📰 正在处理提要: {feed_title} (来源: {source_name})")
    print(f"文章总数: {len(feed.entries)}.")

    extracted_count = 0
    filtered_out_count = 0

    for entry in feed.entries:
        title = entry.get('title', '无标题')
        published_parsed = entry.get('published_parsed')

        # 格式化时间，并获取 UTC datetime 对象用于筛选
        formatted_time, pub_dt_utc = format_published_time(published_parsed, target_timezone_name)
        summary = entry.get('summary', '无简介').strip()
        link = entry.get('link', '#')

        # --- 核心筛选逻辑 ---
        if should_keep_article(source_name, pub_dt_utc):
            articles_data.append({
                "time": formatted_time,
                "published_parsed": published_parsed,
                "source": source_name.split(' (')[0],
                "title": title,
                "summary": summary,
                "link": link
            })
            extracted_count += 1
        else:
            filtered_out_count += 1

    print(f"✅ 成功提取并保留 {extracted_count} 篇符合时间条件的文章。")
    print(f"🗑️ 过滤掉 {filtered_out_count} 篇过期文章。")
    print("-" * 40)

def save_to_json(data, filename=DATA_FILENAME):

        print("✅ 数据已成功存储。")
    except Exception as e:
        print(f"❌ 存储失败: {e}")

def clean_text_formatting(text: str) -> str:
    """
    清理 LLM 文本里可能出现的 markdown 符号等，尽量适配 ReportLab 的 Paragraph HTML 子集。
    """
    if not text:
        return ""

    # 去掉开头 bullet 标记（*, •, -）
    text = re.sub(r"^\s*[\*•-]\s*", "", text)

    # `code` -> <b>code</b>
    text = re.sub(r"`(.*?)`", r"<b>\1</b>", text)

    # {xxx} -> <b>xxx</b>
    text = re.sub(r"\{(.*?)\}", r"<b>\1</b>", text)

    # 去掉 ** (避免残留)
    text = text.replace("**", "")

    return text.strip()

def process_bullet_points(raw_data, style, bullet_char="■", dot_color_hex="#E3120B"):
    """
    把 list[str] 或 str 拆成带 bullet 的 Paragraph 列表。
    bullet_char 可改成 '•' / '■' 等。
    """
    paragraphs = []

    if isinstance(raw_data, list):
        items = raw_data
    elif isinstance(raw_data, str):
        # 允许传入一个带换行的长字符串
        items = [x for x in raw_data.split("\n") if x.strip()]
    else:
        items = []

    for item in items:
        cleaned = clean_text_formatting(item)
        if cleaned:
            dot = f'<font color="{dot_color_hex}">{bullet_char}</font>'
            paragraphs.append(Paragraph(f"{dot} {cleaned}", style))

    return paragraphs

def get_current_date_str_upper():
    now = datetime.now()
    day = now.day
    suffix = "th" if 11 <= day <= 13 else {1: "st", 2: "nd", 3: "rd"}.get(day % 10, "th")
    return now.strftime(f"%B {day}{suffix} %Y").upper()

def get_month_range_str():
    now = datetime.now()
    current_month_name = now.strftime("%B").upper()
    first_day_of_current_month = now.replace(day=1)
    last_day_of_previous_month = first_day_of_current_month - timedelta(days=1)
    previous_month_name = last_day_of_previous_month.strftime("%B").upper()
    return f"{previous_month_name}-{current_month_name}"

def get_filename_date_str():
    return datetime.now().strftime("%m%d%Y")

def create_economist_pdf(data: dict, filename: str):
    logging.info(f"--- 生成 PDF: {filename} ---")

    # 更紧凑的边距（你可以继续缩小，比如 0.25 inch）
    doc = SimpleDocTemplate(
        filename,
        pagesize=letter,
        leftMargin=0.30 * inch,
        rightMargin=0.30 * inch,
        topMargin=0.30 * inch,
        bottomMargin=0.30 * inch,
    )

    story = []
    CONTENT_WIDTH = 7.9 * inch

    # ---------- A. Header ----------
    header_line = Table([[""]], colWidths=[CONTENT_WIDTH], rowHeights=[3])
    header_line.setStyle(TableStyle([("BACKGROUND", (0, 0), (-1, -1), ECO_RED)]))
    story.append(header_line)
    story.append(Spacer(1, 4))

    story.append(Paragraph("<b>THE GLOBAL FINANCE & ECONOMICS WEEKLY BRIEFING</b>", styles["EcoTitle"]))
    story.append(Paragraph(f"PRODUCED ON {get_current_date_str_upper()}", styles["EcoSubtitle"]))
    story.append(Spacer(1, 4))

    # ---------- B. Top two cards ----------
    left_content = []
    left_content.append(Paragraph("72-HOUR MARKET DYNAMICS", styles["CardHeader"]))
    left_content.extend(
        process_bullet_points(
            data.get("market_dynamics_summary", []),
            styles["CardBody"],
            bullet_char="■",
            dot_color_hex="#E3120B",
        )
    )

    right_content = []
    month_range = get_month_range_str()
    right_content.append(Paragraph(f"{month_range} DEEP ANALYSIS", styles["CardHeader"]))
    right_content.extend(
        process_bullet_points(
            data.get("deep_analysis_summary", ""),
            styles["CardBody"],
            bullet_char="■",
            dot_color_hex="#0D4590",
        )
    )

    gap_width = 0.1 * inch
    col_width = (CONTENT_WIDTH - gap_width) / 2

    top_table = Table([[left_content, "", right_content]], colWidths=[col_width, gap_width, col_width])
    top_table.setStyle(TableStyle([
        ("BACKGROUND", (0, 0), (0, 0), ECO_GREY),
        ("VALIGN", (0, 0), (-1, -1), "TOP"),
        ("TOPPADDING", (0, 0), (0, 0), 6),
        ("BOTTOMPADDING", (0, 0), (0, 0), 6),
        ("LEFTPADDING", (0, 0), (0, 0), 6),

        ("RIGHTPADDING", (2, 0), (2, 0), 6),
        ("LINEABOVE", (2, 0), (2, 0), 2, ECO_BLUE),
    ]))
    story.append(top_table)
    story.append(Spacer(1, 8))

    # ---------- C. Category block (auto keys) ----------
    ft_dict = safe_get_dict(data, "market_dynamics_by_category")
    eco_dict = safe_get_dict(data, "deep_analysis_by_category")

    # Debug: print keys for quick diagnosis
    logging.info(f"FT categories found: {list(ft_dict.keys())}")
    logging.info(f"Eco categories found: {list(eco_dict.keys())}")

    all_keys = set(ft_dict.keys()) | set(eco_dict.keys())
    ordered_keys = [k for k in PREFERRED_ORDER if k in all_keys] + sorted([k for k in all_keys if k not in PREFERRED_ORDER])

    categories_flow = []

    for cat_key in ordered_keys:
        ft_text = safe_get_dict(ft_dict, cat_key).get("category_summary", "")
        eco_text = safe_get_dict(eco_dict, cat_key).get("category_summary", "")

        if not ft_text and not eco_text:
            continue

        # 中文显示标题（找不到映射就退回原 key）
        display_zh = CATEGORY_DISPLAY_ZH.get(cat_key, cat_key)
        title_line = f"<b>{display_zh}</b>"

        categories_flow.append(Paragraph(title_line, styles["SectionBar"]))

        if ft_text:
            categories_flow.append(Paragraph(clean_text_formatting(ft_text), styles["SectionBody"]))

        if eco_text:
            if ft_text:
                categories_flow.append(Spacer(1, 4))
            categories_flow.append(Paragraph(clean_text_formatting(eco_text), styles["SectionBody"]))

        categories_flow.append(Spacer(1, 6))

    # Wrap category area with border

            ("BOTTOMPADDING", (0, 0), (-1, -1), 6),
            ("VALIGN", (0, 0), (-1, -1), "TOP"),

        logging.warning("⚠️ 分类区块为空：请检查 JSON 中 *_by_category 的结构是否存在/是否为空。")
